# 这里是我随手写的Pandas API 的一些笔记，为了省事直接拷贝于此！
import numpy as np
import pandas


'''
this is a pandas demo

to show basic API of pandas

to install the lib ...

pip:
    pip install pandas IN terminal
    
or if conda is available in your operating system, using
    conda install pandas == version_you_want_to_install
    
'''

'''
Pandas 1 : using this sentence to show the version of pandas:>
'''
print(pandas.__version__)

# Sections1: Series

'''
Series in pandas is considered as a data structure with index

As D & Alg has already pointed out, index can be string or just numbers
in pandas, it's the same
'''
'''
---------------------------------------------------------------------------------
    the creation of series
---------------------------------------------------------------------------------
'''
print('this is to create a series by using two ways')
''' to check, try to run it in terminal '''

# this is a list
height_among_stu_list = [131, 141, 182, 109, 178]
print(height_among_stu_list)

# this is a series
print("Series is gonna printed:>\n")
height_series = pandas.Series(height_among_stu_list)
print(height_series)

# we can set a name of a series:>

set_name = pandas.Series([1, 2, 3], name='An Increasing subsequence')
print(set_name)

# Back to what I said before... series is a data structure with index... maybe hashSet or hash table?

val = [1, 2, 3, 4, 5]
label = ['a', 'b', 'c', 'd', 'e']
set_with_label = pandas.Series(val, index=label)
print(set_with_label)

# if you are serious about the index list, use series.index to gain a result:>

print(set_with_label.index)


# another quick way to create a series is to use a dictionary as input

dic = {
    'A': 111,
    'B': 222,
    'C': 333,
    'D': 444
}

ser_dic = pandas.Series(dic)

print(ser_dic)

# here comes an interesting questions:> what if we have 3 inputs but owns four labels? have a try...
dic = {
    'A': 111,
    'B': 222,
    'C': 333,
}

ser_dic_loss = pandas.Series(dic, index=['A', 'B', 'C', 'D'])
print(ser_dic_loss)

# we get a nan as result! NAN in pandas means a loss of number. We are supposed to make a deal with it!


# another creation , especially for vals, is to do in this way:>1

set_num = pandas.Series(10,index=['A', 'B', 'C', 'D'])
print(set_num)

# we see the labels own the same value of 10...

'''
-----------------------------------------------------------------------------------
    the usage of index and its relative API
-----------------------------------------------------------------------------------
'''

# usually, when we visit a hash set, we do visit elements by offering a label.
# for example, we can get 'A' by inputting 1 in a hash set:>
# for series in pandas, things got the same...

dic_for_search = dic

setSearch = pandas.Series(dic_for_search)

print("Print the key by the visitor[]")

# this for loop is to visit elements in series
searchLabels = ['A', 'B', 'C']
for i in range(0, 3):
    print(setSearch[searchLabels[i]])

# but if the key is hard to get, while it's relatively easy to know the position,
# this API is just for the case:> called 'iloc': i locations in other words...

for i in range(0, 3):
    print(setSearch.iloc[i])

# differently, we can output the suitable value by just inputting the logical expression in []
print("this key - val set is a set which the val is above 222")
print(setSearch[setSearch > 222])

# Also, if the val is supported to make basic calculations, we can operate all the elements by...

setSearch += 222
print(setSearch)

# here comes a big difference comparing list in python and series in pandas

mylist = [2]
print(mylist)
mylist += [3]
print(mylist)

# See , we actually directly append the elements backward to the mylist

# while this one is...
setSearch += [2]
print(setSearch)

# add the elements to all the vals instead of appending


'''
----------------------------------------------------------------------------------------------
            index
----------------------------------------------------------------------------------------------
'''

# index has variety of types

# when we do not offer a coefficient of index in the initialization, pandas will
# automatically use RangeIndex

val = [1, 2, 3, 4, 5]
a = pandas.Series(val)
b = pandas.Series(val, index=[10, 20, 30, 40, 50])
c = pandas.Series(val, list('abcde'))
print('\n')
print(a.index)
print(b.index)
print(c.index)

''' VAlS in pandas '''

# 1 iloc and loc

# loc is [ ] while iloc is [ )

# no examples... as they have been used previously

print("this is the size:> ", setSearch.size)

# is_unique:> if there is the same elem in the set, return false, else return true

print(setSearch.is_unique)

# we can modify the elements by offering the key name:>

setSearch['A'] = 10000
print(setSearch)

# also it can be multipy:>

setSearch['B':'C'] = 20000
print(setSearch)

# back to the way of appending the series: by just offering a new key with value:>

setSearch['E'] = 2000000
print(setSearch)

# or using the api: append
appender = pandas.Series([10000000,20000000],index=['F', 'G'])
setSearch.append(appender)
print(setSearch)

'''
-----------------------------------------------------------------------------------------------------

                                        del

-----------------------------------------------------------------------------------------------------
'''

# use del directly to erase the elements

del setSearch['A']
print(setSearch)

# or use tha api in pandas:>
setSearch.drop(index='B', inplace=True)
print(setSearch)

# if we want to erase the duplicate elements, use:>
duplicate = pandas.Series([1, 1, 2, 2, 3, 3],index=list('abcdef'))
print(duplicate)
duplicate = duplicate.drop_duplicates(keep='first')
print(duplicate)


# duel and check the NAN

aNAN = pandas.Series([1, np.NaN, 2, np.NaN, 3, np.NaN], list('abcdef'))
print(aNAN.isna())

# use dropna() can be a good choice to erase the invalid data
aNAN = aNAN.dropna()
print(aNAN)
